\begin{introduction}{Basic laws for electrical circuits.}%
Monday, January 7\hypertarget{p-1}{}%
This two-page worksheet was generously donated to the sample article by Virgil Pierce at a CuratedCourses workshop in August~2018.  It has default (skinny) left and right margins, but we have specified longer top and bottom margins, with the top being the larger of the two.%
\begin{theorem}{Ohms Law.}{}{theorem-1}%
\hypertarget{p-2}{}%
The current through a resistor is proportional to the ratio of the \emph{Voltage} to the \emph{Resistance}%
\begin{equation*}
I = \frac{V}{R} 
\end{equation*}
Or for our purposes%
\begin{equation*}
I R = V 
\end{equation*}
%
\end{theorem}
\begin{theorem}{Kirchoffs Current Law.}{}{theorem-2}%
\hypertarget{p-3}{}%
The sum of the currents in a network meeting at a point is zero.%
\begin{equation*}
\sum_{k=1}^n I_k = 0 
\end{equation*}
%
\end{theorem}
\begin{example}{Kirchoff's Current Law.}{example-1}%
\hypertarget{p-4}{}%
For the circuit below \(I_1 + I_2 = I_3 \).%

\end{tikzpicture}
}
}
\end{sbspanel}%
\end{sidebyside}%
\end{example}
\begin{theorem}{Kirchoffs Voltage Law.}{}{theorem-3}%
\hypertarget{p-5}{}%
The sum of the voltages around any closed circuit (or subcircuit) is zero.%
\begin{equation*}
\sum_{k=1}^n V_k = 0 
\end{equation*}
%
\end{theorem}
\hypertarget{p-6}{}%
Kirchoffs Current Law and Kirkoffs Voltage Law combined with Ohms Law gives for any circuit of resistors and sources a linear system that may (or may not) determine the currents.%
\end{introduction}%
